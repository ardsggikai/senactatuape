11-03-2022

SP
enable 
 configure terminal
 hostname SP
 interface fast 0/0
 description porta lan do SP
  no shutdown
  ip add 192.168.100.1 255.255.255.0
  exit
 interface s1/0
 description Porta WAN que liga SP em RJ
 no shutdown
 ip add 20.0.0.1 255.0.0.0
 clock rate 128000 
 bandwidth 128
 exit
router rip
 net 20.0.0.0
 net 192.168.100.0
 exit
 exit
copy run star
