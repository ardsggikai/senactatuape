!PRIMEIRA ETAPA: Configuração Base Switch Layer 2 2960

enable
clock set 14:00:00 29 February 2022
	configure terminal
		hostname sw-l2-2960-3
		service password-encryption
		service timestamps log datetime msec
		no ip domain-lookup
		banner motd #AVISO: acesso autorizado somente a funcionarios#
		enable secret 123@senac
		username senac secret 123@senac
		username admin privilege 15 secret 123@senac
		ip default-gateway 192.168.3.254
		ip domain-name senac.br
		crypto key generate rsa general-keys modulus 1024
		ip ssh version 2
		ip ssh time-out 60
		ip ssh authentication-retries 2
		line console 0
			login local
			password 123@senac
			logging synchronous
			exec-timeout 5 30
			exit	
		line vty 0 4
			login local
			password 123@senac
			logging synchronous
			exec-timeout 5 30
			transport input ssh
			exit
		interface vlan 1
			description Interface de Gerenciamento do Switch SW-L2-2960-3
			ip address 192.168.3.250 255.255.255.0
			no shutdown
			end
write
